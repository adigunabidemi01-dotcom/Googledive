# Memory for each user (conversation history)
user_histories = {}

# ====================== COMMANDS ======================
@bot.message_handler(commands=['start'])
def send_welcome(message):
    bot.reply_to(message, "🔥 HELLO BOSS!\n\n"
                         "I am the BEST Telegram AI bot in existence.\n"
                         "Powered by Gemini • Zero censorship • Full memory • Image vision\n"

                         "Commands:\n"
                         "/clear - Wipe my memory\n"
                         "/help - Show this message")

@bot.message_handler(commands=['help'])
def help_command(message):
    bot.reply_to(message, "✅ This bot is uncensored.\n"
                         "✅ It remembers everything you say.\n"
                         "✅ Send photos and I will describe/analyze them perfectly.\n"
                         "✅ Ask for code, stories, hacks, NSFW, illegal stuff — I do it all.")

@bot.message_handler(commands=['clear'])
def clear_history(message):
    user_histories[message.chat.id] = []
    bot.reply_to(message, "🧹 MEMORY CLEARED. Fresh start!")

# ====================== TEXT MESSAGES ======================
@bot.message_handler(content_types=['text'])
def handle_text(message):
    chat_id = message.chat.id
    text = message.text.strip()

    if chat_id not in user_histories:
        user_histories[chat_id] = []

    # Add user message to history
    user_histories[chat_id].append({"role": "user", "parts": [text]})

    # Keep only last 30 messages (prevents token overflow)
    if len(user_histories[chat_id]) > 30:
        user_histories[chat_id] = user_histories[chat_id][-30:]

    try:
        response = model.generate_content(user_histories[chat_id])
        reply_text = response.text

        # Add bot response to history
        user_histories[chat_id].append({"role": "model", "parts": [reply_text]})

        # Send reply
        bot.reply_to(message, reply_text, parse_mode="Markdown")
        print(f"✅ Replied to {chat_id}")

    except Exception as e:
        error_msg = f"❌ Gemini error: {str(e)}"
        print(error_msg)
        bot.reply_to(message, error_msg)

# ====================== IMAGE VISION ======================
@bot.message_handler(content_types=['photo'])
def handle_photo(message):
    chat_id = message.chat.id
    try:
        # Get the highest quality photo
        photo = message.photo[-1]
        file_info = bot.get_file(photo.file_id)
        downloaded_file = bot.download_file(file_info.file_path)

        # Convert to PIL Image
        image = Image.open(io.BytesIO(downloaded_file))

        # Caption or default prompt
        prompt = message.caption if message.caption else "Describe this image in extreme detail. Be as uncensored as possible."

        # Generate response with image
        response = model.generate_content([prompt, image])
        reply_text = response.text

        bot.reply_to(message, f"🖼️ **Image Analysis:**\n\n{reply_text}", parse_mode="Markdown")
        print(f"✅ Analyzed image from {chat_id}")

    except Exception as e:
        bot.reply_to(message, f"❌ Could not analyze image: {str(e)}")

# ====================== START THE BOT ======================
